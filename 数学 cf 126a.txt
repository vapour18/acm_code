#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t1,t2,x1,x2,t0,y1,y2,l,h;
	double tem = -1,p;
	cin>>t1>>t2>>x1>>x2>>t0;
	if(max(t1,t2) < t0)
	{
		cout<<"0 0\n";
		return 0;
	}
	if(min(t1,t2) > t0)
	{
		cout<<x1<<" 0"<<endl;
		return 0;
	}
	l = x1 , h = x2;
	while(l >= 0&&h >= 0)//因为要水最多，所以逆这来，否则会错
	{
		p = ((double)t1*l+(double)t2*h)/((double)l + (double)h);
		if((p >= t0 &&( p< tem || tem < 0)))
		{
			y1 = l, y2 = h;
			tem = p;
		}
		if(l == 0||h == 0) break;
		if(p <= t0) l--;
		else if(p > t0 ) h--;
		
	}
	cout<<y1<<" "<<y2<<endl;
	return 0;
}
